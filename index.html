Bionic Hand 프로젝트 블로그

MYOWARE 2.0을 이용한 근전도 기반 의수 - 엄지 제어

📎 프로젝트 개요

본 프로젝트는 MYOWARE 2.0 센서를 통해 손가락 중 엄지를 제어하는 Bionic Hand 의수 장치를 만드는 것을 목표로 하였습니다.

PPT 보러가기

🧠 아두이노 코드


#define THRESHOLD 300
int sensorPin = A0;
int ledPin = 9;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int emgValue = analogRead(sensorPin);
  Serial.println(emgValue);
  if (emgValue > THRESHOLD) {
    digitalWrite(ledPin, HIGH);
  } else {
    digitalWrite(ledPin, LOW);
  }
  delay(50);
}
