MYOWARE 2.0을 이용한 엄지 근전도 제어 의수

프로젝트 소개

본 프로젝트는 MYOWARE 2.0 근전도 센서를 이용해 손가락(특히 엄지)의 움직임을 감지하여 인공 손을 제어하는 시스템입니다. 고등학교 동아리 활동으로 진행되었으며, 아두이노를 활용한 프로그래밍과 하드웨어 제어를 통해 제작하였습니다.

예시 코드 (Arduino)

// MYOWARE 2.0을 이용한 엄지 감지 예시 코드
int emgPin = A0;
int threshold = 500;

void setup() {
  Serial.begin(9600);
  pinMode(9, OUTPUT); // 서보 또는 LED 연결
}

void loop() {
  int emg = analogRead(emgPin);
  if (emg > threshold) {
    digitalWrite(9, HIGH); // 근육 수축 시 동작
    Serial.println("엄지 감지됨");
  } else {
    digitalWrite(9, LOW);
  }
  delay(100);
}
